
// --- Ejemplo de error: Uso de variable fuera de su scope ---
CREAR NUMERO x = 10;
IF (x == 10) {
    CREAR TEXTO mensaje = "Dentro del IF";
    MOSTRAR mensaje;
}
// Error DSL(301): 'mensaje' ya no existe fuera del IF
MOSTRAR mensaje;

// --- Ejemplo funcional ---
CREAR NUMERO x = 10;
IF (x == 10) {
    CREAR TEXTO mensaje = "Dentro del IF";
    MOSTRAR mensaje;
}
// No se usa 'mensaje' fuera del IF

// --- Ejemplo de error: Operación matemática entre estructuras incompatibles ---
CREAR COLA miCola;
CREAR NUMERO resultado = miCola + 5; // Error DSL(305)

// --- Ejemplo funcional ---
CREAR COLA miCola;
ENCOLAR 5 EN miCola;
CREAR NUMERO resultado = 5 + 5;

// --- Ejemplo de error: Asignación de tipo incorrecto ---
CREAR TEXTO t = 123; // Error DSL(303)
CREAR NUMERO n = "texto"; // Error DSL(303)

// --- Ejemplo funcional ---
CREAR TEXTO t = "hola";
CREAR NUMERO n = 123;

// --- Ejemplo de error: Uso de comando de ARBOL en LISTA ---
CREAR LISTA_ENLAZADA miLista;
AGREGARNODO 10 EN miLista; // Error DSL(304)

// --- Ejemplo funcional ---
CREAR LISTA_ENLAZADA miLista;
INSERTAR_INICIO 10 EN miLista;

// --- Ejemplo de error: Uso de variable no inicializada ---
CREAR NUMERO a;
MOSTRAR a; // Error DSL(306)

// --- Ejemplo funcional ---
CREAR NUMERO a = 5;
MOSTRAR a;

// --- Ejemplo de error: Redeclaración en el mismo ámbito ---
CREAR TEXTO var = "uno";
CREAR TEXTO var = "dos"; // Error DSL(302)

// --- Ejemplo funcional ---
CREAR TEXTO var = "uno";
MOSTRAR var;

// --- Test de Errores Cruzados entre Estructuras ---

CREAR GRAFO miGrafo;
CREAR ARBOL_BINARIO miArbol;
CREAR TABLA_HASH miHash;

// Error DSL(304): Usando comando de ARBOL ('AGREGARNODO') en un GRAFO.
AGREGARNODO 10 EN miGrafo;

// Error DSL(304): Usando comando de GRAFO ('BFS') en una TABLA_HASH.
BFS EN miHash;

// Error DSL(304): Usando comando de TABLA_HASH ('REHASH') en un ARBOL.
REHASH EN miArbol;

// Error DSL(305): Comparando estructuras complejas.
IF (miGrafo == miArbol) {
    MOSTRAR "Esto no tiene sentido y debería ser un error.";
